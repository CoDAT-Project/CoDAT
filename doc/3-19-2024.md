//==================================================================================================================================================//

// \s+ is one or more spaces
// \w+ is one or more words
private static final Pattern PATTERN = Pattern.compile("@LinkAnnotation\\s+(\\w+)");

//==================================================================================================================================================//

// PsiSearchHelper is a class and PsiSearchHelper.getInstance is a method that creates an instance of PsiSearchHelper, which is a utility class, i.e. PsiSearchHelper is a class name.
PsiSearchHelper searchHelper = PsiSearchHelper.getInstance(anyElement.getProject());

//==================================================================================================================================================//

// annotationMap maps comment annotation (a string) to the List<PsiComment> object which contains the actual comment.
Map<String, List<PsiComment>> annotationsMap = new HashMap<>();

//==================================================================================================================================================//

 searchHelper.processAllFilesWithWordInComments("@LinkAnnotation", GlobalSearchScope.allScope(anyElement.getProject()), file -> {
    // Execute this code on each file that is found
    processFileForComments(file, annotationsMap);
    return true;
});
